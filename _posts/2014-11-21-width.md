### Set the width of an element

#### jQuery
{% highlight javascript %}
$(element).width(30);
{% endhighlight %}

#### Native JavaScript
{% highlight javascript %}
element.style.width = "30px";
{% endhighlight %}

### Get the width of an element

#### jQuery
{% highlight javascript %}
var element_width = $(element).width();
{% endhighlight %}

#### Native JavaScript
{% highlight javascript %}
var element_width = element.style.width;
{% endhighlight %}
