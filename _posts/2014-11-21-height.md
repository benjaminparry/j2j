### Set the height of an element

#### jQuery
{% highlight javascript %}
$(element).height(30);
{% endhighlight %}

#### Native JavaScript
{% highlight javascript %}
element.style.height = "30px";
{% endhighlight %}

### Get the height of an element

#### jQuery
{% highlight javascript %}
var element_height = $(element).height();
{% endhighlight %}

#### Native JavaScript
{% highlight javascript %}
var element_height = element.style.height;
{% endhighlight %}
